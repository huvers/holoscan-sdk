cmake_minimum_required(VERSION 3.20)
project(multi_port_or_combiner CXX)

# Finds the package holoscan
find_package(holoscan REQUIRED CONFIG
             PATHS "/opt/nvidia/holoscan" "/workspace/holoscan-sdk/install")

add_executable(multi_port_or_combiner
  multi_port_or_combiner.cpp
)
target_link_libraries(multi_port_or_combiner
  PRIVATE
  holoscan::core
  holoscan::ops::ping_tx
)

# Testing
if(BUILD_TESTING)
  add_test(NAME EXAMPLE_CPP_MULTI_PORT_OR_COMBINER_TEST
           COMMAND multi_port_or_combiner
           WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
          )
  set_tests_properties(
    EXAMPLE_CPP_MULTI_PORT_OR_COMBINER_TEST PROPERTIES
    # count > 10 due to the OR condition
    PASS_REGULAR_EXPRESSION "count 15, message received"
    FAIL_REGULAR_EXPRESSION "count 21, message received"
  )
endif()

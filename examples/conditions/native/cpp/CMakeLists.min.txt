cmake_minimum_required(VERSION 3.20)
project(ping_period_native CXX)

# Finds the package holoscan
find_package(holoscan REQUIRED CONFIG
             PATHS "/opt/nvidia/holoscan" "/workspace/holoscan-sdk/install")

add_executable(ping_periodic_native
  ping_periodic_native.cpp
)

target_link_libraries(ping_periodic_native
  PRIVATE
  holoscan::core
  holoscan::ops::ping_rx
  holoscan::ops::ping_tx
)

# Create examples
add_executable(message_available_native
  message_available_native.cpp
)
target_link_libraries(message_available_native
  PRIVATE
  holoscan::core
  holoscan::ops::ping_rx
  holoscan::ops::ping_tx
)
# Testing
if(BUILD_TESTING)
  add_test(NAME EXAMPLE_CPP_PING_PERIODIC_NATIVE_TEST
           COMMAND ping_periodic_native
           WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
          )
  set_tests_properties(EXAMPLE_CPP_PING_PERIODIC_NATIVE_TEST PROPERTIES
                       PASS_REGULAR_EXPRESSION "Rx message value: 5")

  add_test(NAME EXAMPLE_CPP_MESSAGE_AVAILABLE_NATIVE_TEST
           COMMAND message_available_native
           WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
          )
  set_tests_properties(EXAMPLE_CPP_MESSAGE_AVAILABLE_NATIVE_TEST PROPERTIES
                       ENVIRONMENT "HOLOSCAN_LOG_LEVEL=DEBUG"
                       PASS_REGULAR_EXPRESSION "Not adding default condition to port 'in' of operator 'rx'"
                       PASS_REGULAR_EXPRESSION "Rx message value: 10"
                       FAIL_REGULAR_EXPRESSION "Operator 'rx' failed to receive message")
endif()

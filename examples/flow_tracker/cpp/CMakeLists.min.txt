cmake_minimum_required(VERSION 3.20)
project(holoscan_flow_tracker CXX)

# Finds the package holoscan
find_package(holoscan REQUIRED CONFIG
             PATHS "/opt/nvidia/holoscan" "/workspace/holoscan-sdk/install")

add_executable(flow_tracker
  flow_tracker.cpp
)

target_link_libraries(flow_tracker
  PRIVATE
  holoscan::core
)

# Testing
if(BUILD_TESTING)
  add_test(NAME EXAMPLE_CPP_FLOW_TRACKER_TEST
           COMMAND flow_tracker
           WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
          )
  set_tests_properties(EXAMPLE_CPP_FLOW_TRACKER_TEST PROPERTIES
                       PASS_REGULAR_EXPRESSION "Data Flow Tracking Results"
                       PASS_REGULAR_EXPRESSION "Path 1: root1,middle1,leaf1")
endif()

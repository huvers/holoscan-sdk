cmake_minimum_required(VERSION 3.20)
project(conditional_routing CXX)

# Finds the package holoscan
find_package(holoscan REQUIRED CONFIG
             PATHS "/opt/nvidia/holoscan" "/workspace/holoscan-sdk/install")

add_executable(conditional_routing_execution
  conditional_routing_execution.cpp
)

target_link_libraries(conditional_routing_execution
  PRIVATE
  holoscan::core
)

# Testing
if(BUILD_TESTING)
  add_test(NAME EXAMPLE_CPP_CONDITIONAL_ROUTING_EXECUTION_TEST
           COMMAND conditional_routing_execution
           WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
          )
  set_tests_properties(EXAMPLE_CPP_CONDITIONAL_ROUTING_EXECUTION_TEST PROPERTIES
                       PASS_REGULAR_EXPRESSION ".*Sending value 1 - node1.*Received value 1 - node2.*Sending value 2 - node1.*Received value 2 - node3.*"
                       FAIL_REGULAR_EXPRESSION "error"
                      )
endif()
